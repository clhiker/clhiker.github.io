title: 桶排序

/*

  • 桶排序
  • 排序的关键是选取合适的映射函数，使得桶中的数据相对均匀
  • 2019,3,8

  • by 最黑的黑客
    /
    //eg. 假设一组数据(20长度)为
    //[63,157,189,51,101,47,141,121,157,156,194,117,98,139,67,133,181,13,28,109]
    //现在需要按5个分桶，进行桶排序，实现步骤如下:
    //找到数组中的最大值194和最小值13，然后根据桶数为5，计算出每个桶中的数据范围为(194-13+1)/5=36.4
    //遍历原始数据，(以第一个数据63为例)先找到该数据对应的桶序列Math.floor(63 - 13) / 36.4) =1，然后将该数据放入序列为1的桶中(从0开始算)
    //当向同一个序列的桶中第二次插入数据时，判断桶中已存在的数字与新插入的数字的大小，按从左到右，从小打大的顺序插入。如第一个桶已经有了63，再插入51，67后，桶中的排序为(51,63,67) 一般通过链表来存放桶中数据，但js中可以使用数组来模拟
    //全部数据装桶完毕后，按序列，从小到大合并所有非空的桶(如0,1,2,3,4桶)
    //合并完之后就是已经排完序的数据

    #include <stdio.h>
    #include <stdlib.h>
    #include <time.h>

    // 借助快排排序桶中元素
    void quickSort(int*a, int n){

    // 特殊情况提前处理，减少工作量
if (n <= 1)
    return;
if (n == 2){
    if (a[0] < a[1])
        return;
    else{
        int temp = a[0];
        a[0] = a[1];
        a[1] = temp;
        return;
    }
}

int i=0, j=n-1;
int flag = i;
while(i != j) {
    if (a[j] < a[flag]){
        while (i != j){
            if (a[i] > a[flag]){
                // 对于符合条件的两个数进行置换
                int temp = a[i];
                a[i] = a[j];
                a[j] = temp;
                break;
            }
            i++;
        }
        if (i == j)
            break;  // 注意临界条件
    }
    j--;    // 基准是左边的数，从右边开始走更好
}
// 最后要把flag位和结束位置换
// 注意 9 11 36 和 11 9 36的区别
int temp = a[flag];
a[flag] = a[i];
a[i] = temp;

    int *right = (int*)malloc(sizeof(int) * (n - i - 1));
    for(int count1=i+1, count2=0; count1 < n; count1++, count2++){
        right[count2] = a[count1];
    }
    quickSort(a, i);
    quickSort(right, n-i-1);
    for(int count1=i+1, count2=0; count1 < n; count1++, count2++){
        a[count1] = right[count2];
    }
    // 不要忘记释放
    free(right);
}

// 统计重复次数
void bucketSort(int*a, int n){
    int max_num = a[0];
    int min_num = a[0];
    for(int i=0;i<n;i++){
        if (max_num < a[i])
            max_num = a[i];
        if (min_num > a[i])
            min_num = a[i];
    }

    // 计算间隔数，为数据入桶做铺垫
    int interval = (max_num - min_num) / n + 1; // 避免0的出现

    // 假设桶宽为10；待排数据中没有负数
    // 这里选取简单的映射函数,假设有n个桶
    // 桶的二位数组
    int row = n;    // n个桶
    int column = n;    // 每个桶中最多能装n个数据
    int **bucket = malloc(sizeof(int*) * row);
    for(int i=0;i<row;i++){
        bucket[i] = malloc(sizeof(int) * column);
    }
    // 使用-1填充桶
    for (int i=0;i<row;i++) {
        for(int j=0;j<column;j++)
            bucket[i][j] = -1;
    }

    // 还需要一个数组来记录桶中已经放入的元素个数（链表则不需要考虑这件事）
    int *counter = (int*) malloc(sizeof(int) * n);
    for(int i=0;i<n;i++)
        counter[i] = 0;

    // 数据进入桶中
    for(int i=0; i<n; i++) {
        int index = (a[i] - min_num) / interval;
        bucket[index][counter[index]] = a[i];
        counter[index] += 1;
    }

    // 排序桶中元素
    for (int i=0;i<row;i++) {
        int temp = 0;
        for(int j=0;j<column;j++){
            if(bucket[i][j] != -1)
                temp++;
        }
        if(temp != 0)
            quickSort(bucket[i], temp);
    }

    int k = 0;
    for (int i=0;i<row;i++) {
        for(int j=0;j<column;j++)
            if(bucket[i][j] != -1){
                a[k++] = bucket[i][j];
            }
    }

    free(bucket);
    free(counter);
}

int main() {

    int n = 10;
    int *a = (int*)malloc(sizeof(int) * n);
    srand(time(NULL));
    for (int i = 0; i< n; i++){
        a[i] = rand()%100;
    }
    for(int i=0; i<n; i++){
        printf("%d\t", a[i]);
    }
    printf("\n");

    // 数字范围（假设已知间隔为1；
    bucketSort(a, n) ;

    for(int i=0; i<n; i++){
        printf("%d\t", a[i]);
    }
    free(a);
    return 0;
}
